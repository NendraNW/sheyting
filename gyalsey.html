རྒྱལ་བའི་སྲས་ཀྱི་ལག་ལེན་ སུམ་ཅུ་སོ་བདུན་མ་ཟེར་བའི་ འགྲེལ་བཤད་བསྡུས་པ། བདུད་བཞི་ལས་རྒྱལ་བའི་ རྫོགས་པའི་སངས་རྒྱས་དང་ ཐུགས་ཀྱི་སྲས་བྱང་ཆུབ་སེམས་དཔའ་ཚུ་ གིས་ གཞན་སེམས་ཅན་གྱི་དོན་མཛད་པའི་སྐབས་ བསྒྲུབ་བྱའི་ ལག་ལེན་བྱང་ཆུབ་སེམས་དཔའི་ སྤྱོད་པ་ རྒྱ་མཚོ་བཟུམ་སྦེ་ཡོད་པའི་ནང་ལས་ བཅུད་བསྡུས་ཏེ་ ཤོ་ལོ་ཀ་སུམ་ཅུ་སོ་བདུན་ འདི་ ནང་ལུ་བཀོད་དེ་ བཞུགས་ཡོད་ཟེར་བའི་དོན་ཨིན།
